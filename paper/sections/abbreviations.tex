\section{Abbreviations}\label{sec:abbreviations}

\begin{description}
  \item[BCEA] Bayesian cost-effectiveness analysis.
  \item[CEAF] Cost-effectiveness acceptability frontier.
  \item[ENBS] Expected net benefit of sampling.
  \item[EVPI] Expected value of perfect information.
  \item[EVPPI] Expected value of partial perfect information.
  \item[EVSI] Expected value of sample information.
  \item[QALY] Quality-adjusted life-year.
  \item[VOI] Value of Information\@.
\end{description}
